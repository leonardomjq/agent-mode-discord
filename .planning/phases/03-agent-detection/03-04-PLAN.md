import type { Event } from "../state/types";
import { createShellIntegrationDetector } from "./shellIntegration";
import { createSessionFilesDetector } from "./sessionFiles";
import { createPollingDetector } from "./polling";

export interface DetectorsOrchestratorOptions {
  /** Phase 4 wires detect.customPatterns. v0.1 default = undefined (built-ins only). */
  customPatterns?: Record<string, string[]>;
  /** Phase 4 wires detect.polling.terminalNamePatterns. v0.1 default = empty (no polling). */
  pollingPatterns?: string[];
  /** Phase 4 wires detect.sessionFileStalenessSeconds. v0.1 default = 60. */
  sessionFileStalenessSeconds?: number;

  factories?: {
    shellIntegration?: typeof createShellIntegrationDetector;
    sessionFiles?: typeof createSessionFilesDetector;
    polling?: typeof createPollingDetector;
  };
}

/**
 * Starts all detectors in tier order, dedupes cross-tier signals, and dispatches
 * a SINGLE agent-started/agent-ended sequence to the parent dispatch function.
 *
 * Tier precedence (highest fidelity first):
 *   tier 1 — companion (RESERVED for Phase 5; not wired here)
 *   tier 2 — shellIntegration
 *   tier 3 — sessionFiles
 *   tier 4 — polling
 *
 * Aggregate rules:
 *   - agent-started fires when the orchestrator's "any active" state goes false→true.
 *   - agent-ended fires when "any active" goes true→false.
 *   - When the highest-tier active source changes its agent label, a new agent-started
 *     fires with the new label (no agent-ended in between — reducer treats this as
 *     intra-AGENT_ACTIVE field update per 02-01-SUMMARY).
 */
export function createDetectorsOrchestrator(
  dispatch: (event: Event) => void,
  opts?: DetectorsOrchestratorOptions,
): vscode.Disposable;

  <behavior>
    `createDetectorsOrchestrator(dispatch, opts)` constructs three child detectors and wires each to a tier-aware intercept. The intercept records per-tier { active, agent, lastActivityAt } and recomputes the aggregate after every event.

    **Per-tier intercept:**
    Each child detector is started with a TIER-LOCAL dispatch function instead of the orchestrator's dispatch directly. The local dispatch:
    - On agent-started: `tierStates.set(tier, { active: true, agent, lastActivityAt: Date.now() })`. Then `recomputeAndDispatch()`.
    - On agent-ended: `tierStates.set(tier, { active: false, agent: undefined, lastActivityAt: 0 })`. Then `recomputeAndDispatch()`.
    - Other event types (branch-changed, editor-*) are NOT produced by detectors — pass them through unchanged (defensive forwarding).

    **recomputeAndDispatch():**
    ```typescript
    function recomputeAndDispatch(): void {
      // Find highest-tier active state
      let chosen: { tier: number; agent: string; lastActivityAt: number } | undefined;
      for (const tier of [2, 3, 4] as const) {
        const s = tierStates.get(tier);
        if (s?.active && s.agent) {
          chosen = { tier, agent: s.agent, lastActivityAt: s.lastActivityAt };
          break; // first hit = highest fidelity
        }
      }
      const wasActive = aggregateActive;
      const isActive = chosen !== undefined;
      aggregateActive = isActive;
      if (!wasActive && isActive) {
        dispatch({ type: "agent-started", agent: chosen!.agent });
        lastDispatchedAgent = chosen!.agent;
      } else if (wasActive && !isActive && lastDispatchedAgent) {
        dispatch({ type: "agent-ended", agent: lastDispatchedAgent });
        lastDispatchedAgent = undefined;
      } else if (isActive && chosen!.agent !== lastDispatchedAgent) {
        // Highest-tier source changed agent — emit new start (reducer treats as intra-AGENT_ACTIVE field update)
        dispatch({ type: "agent-started", agent: chosen!.agent });
        lastDispatchedAgent = chosen!.agent;
      }
    }
    ```

    **Lower-tier suppression at log level:**
    When a tier is active but a higher tier is also active for the same conceptual session, the lower tier's dispatch is "suppressed" purely by virtue of `recomputeAndDispatch` choosing the higher tier first. The lower tier's state IS still recorded in tierStates — it just doesn't influence the dispatched agent label. This is the "lower tiers log at debug only when a higher tier has signal" rule (DET-07). For Phase 3 we don't actually emit debug logs (Phase 4 wires `debug.verbose`); the suppression behavior is structural.

<pitfalls>
- **Reducer interaction (READ 02-01-SUMMARY):** The reducer treats `agent-started` with a different agent in AGENT_ACTIVE as a field update — startTimestamp does NOT reset. Same-agent agent-started returns reference equality (no-op). The orchestrator's "emit new agent-started on label change" matches this; do NOT emit agent-ended → agent-started pairs on label change (would erroneously reset startTimestamp).
- **Tier order is hardcoded:** [2, 3, 4]. When Phase 5 adds tier 1 (companion), the array becomes [1, 2, 3, 4]. This is fine — the loop pattern handles it.
- **Defensive forwarding:** Detectors might (incorrectly) dispatch event types they shouldn't. The orchestrator's per-tier intercept should switch on event.type — only handle agent-started/agent-ended specially; everything else either passes through or is dropped. Choose pass-through for defensiveness.
- **Race: child detector dispatches during orchestrator dispose:** the orchestrator's dispatch closure may still be invoked between dispose() starting and child disposables completing. The parent driver's `shuttingDown` flag in extension.ts handles this — dispatch returns early. The orchestrator does not need its own shuttingDown flag.
- **Factory injection for tests:** opts.factories MUST default to the real factories. Failure mode if you skip the `?? createShellIntegrationDetector` fallback: production code crashes because no factory provided.
- **DO NOT make this orchestrator stateful per-terminal:** that's the child detectors' job (sessionFiles tracks per-file, polling tracks per-terminal, shellIntegration tracks per-terminal). The orchestrator only tracks PER-TIER aggregate state. Leaking per-terminal state up here would duplicate work and cause inconsistencies.
- **Phase 4 hook:** the opts shape includes customPatterns / pollingPatterns / sessionFileStalenessSeconds — Phase 4 will read these from VS Code config and pass them in. Phase 3 calls with no opts; defaults kick in.
</pitfalls>
