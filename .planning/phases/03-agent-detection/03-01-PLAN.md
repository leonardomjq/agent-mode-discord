```typescript
import * as vscode from "vscode";
import type { Event } from "../state/types";
import { buildMatcher, normalizeCommandLine } from "./regex";

/** Per-terminal session tracking (CONTEXT D-04). */
export interface TerminalSession {
  agent: string;
  signalTier: 2;
  lastActivityAt: number;
  /** ms-since-epoch at which grace expires; null when actively running. */
  graceExpiresAt: number | null;
}

export interface ShellIntegrationDetectorOptions {
  /** Override for tests. Defaults to ms-since-epoch via Date.now. */
  now?: () => number;
  /** Override for tests. Defaults to 2000. */
  holdoffMs?: number;
  /** Override for tests. Defaults to 30000. */
  graceMs?: number;
  /** Override for tests — Phase 4 will pipe in detect.customPatterns. Default: built-ins only. */
  customPatterns?: Record<string, string[]>;
  /** Override for tests — defaults to global setTimeout/clearTimeout. */
  setTimeout?: typeof globalThis.setTimeout;
  clearTimeout?: typeof globalThis.clearTimeout;
}

export interface ShellIntegrationDetector {
  readonly tier: 2;
  start(dispatch: (event: Event) => void): vscode.Disposable;
}

export function createShellIntegrationDetector(
  opts?: ShellIntegrationDetectorOptions,
): ShellIntegrationDetector;
```

    ```typescript
    function recomputeAggregate(): void {
      // Active = any session whose graceExpiresAt is null OR > now().
      const now_ = now();
      let active = false;
      let topAgent: string | undefined;
      let topActivity = 0;
      for (const session of sessions.values()) {
        const isActive = session.graceExpiresAt === null || now_ < session.graceExpiresAt;
        if (!isActive) continue;
        active = true;
        if (session.lastActivityAt > topActivity) {
          topActivity = session.lastActivityAt;
          topAgent = session.agent;
        }
      }
      const wasActive = aggregateActive;
      aggregateActive = active;
      if (!wasActive && active && topAgent) {
        dispatch({ type: "agent-started", agent: topAgent });
        lastDispatchedAgent = topAgent;
      } else if (wasActive && !active && lastDispatchedAgent) {
        dispatch({ type: "agent-ended", agent: lastDispatchedAgent });
        lastDispatchedAgent = undefined;
      } else if (active && topAgent && topAgent !== lastDispatchedAgent) {
        // Agent label changed (different agent now highest by lastActivityAt) — dispatch start with new label
        dispatch({ type: "agent-started", agent: topAgent });
        lastDispatchedAgent = topAgent;
      }
    }
    ```

<pitfalls>
- **DET-08 (the most important pitfall in this whole phase):** subscribe to onDidChangeTerminalShellIntegration GLOBALLY (extension lifetime), not per-terminal. Without this, the first command in a fresh terminal where integration activates async is missed. Test #3 enforces this.
- **Pitfall 1 (RESEARCH):** seed existing terminals at start(). onDidChangeTerminalShellIntegration only fires for FUTURE activations. Test #4 enforces this.
- **Pitfall 2 (RESEARCH):** when a new agent-started fires for a terminal in grace, CANCEL the grace timer first. Otherwise the timer fires and dispatches a stale agent-ended even though the terminal is now actively running. Test #8 enforces this.
- **Pitfall 3 (RESEARCH):** onDidCloseTerminal supersedes onDidEndTerminalShellExecution. No grace on close — the terminal is gone, nothing to preserve. Test #10 enforces this.
- **Pitfall 6 (RESEARCH):** commandLine.value may be more accurate at end than at start. We accept the start value for matching agent presence; the end handler re-normalizes to confirm the same command (avoids agent-end dispatch for unrelated commands ending in a terminal where claude was previously running).
- **Pitfall: aggregate dispatch on label change:** when two different agents are running in two terminals, the label dispatched is the highest-tier (here always 2) + most recent lastActivityAt. The aggregate dispatcher must emit a NEW agent-started when the label changes (test #12). This is what the reducer's intra-AGENT_ACTIVE field-update behavior expects (see 02-01-SUMMARY).
- **DO NOT duplicate ANSI strip:** import { normalizeCommandLine } from "./regex". The pure helper is the single source of truth for the strip pipeline. If you re-implement it here, future fixes to one don't propagate to the other.
- **D-18 try/catch:** every onDidX handler body is wrapped. A throwing handler must not crash the extension host. Failure mode if you skip: one bad commandLine.value crashes detection for all subsequent terminals.
</pitfalls>
