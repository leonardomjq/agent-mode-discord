```typescript
// src/detectors/regex.ts — PURE-CORE: no vscode runtime import (D-16, enforced by 03-00)

/** Known v0.1 agents. Custom patterns may produce arbitrary string labels (DET-10). */
export type BuiltInAgent = "claude" | "aider" | "codex" | "gemini" | "opencode";
export type AgentName = BuiltInAgent | string;

/** Built-in agent → array of regex patterns. Each anchored with ^ (first-word). */
export const BUILT_IN_PATTERNS: Record<BuiltInAgent, RegExp[]>;

/** Match result. */
export interface AgentMatch {
  agent: AgentName;
}

/** ANSI CSI regex — exported so shellIntegration.ts reuses the same pure helper. */
export const ANSI_CSI_RE: RegExp;

/**
 * Strip ANSI CSI + normalize. Pure helper. Returns the input unchanged if the
 * input contains no escape sequences. Use only when the calling adapter has
 * already determined confidence === Low.
 */
export function stripAndNormalize(raw: string): string;

/**
 * Normalize a vscode.TerminalShellExecutionCommandLine-shaped object.
 * Confidence enum values: Low=0, Medium=1, High=2 (per @types/vscode 1.115).
 *
 * - Low: full strip pipeline (ANSI strip + line-ending normalize + leading whitespace + prompt prefix + multi-space collapse).
 * - Medium / High: trim only.
 *
 * Accepts a structural shape, NOT vscode.TerminalShellExecutionCommandLine directly,
 * so the module stays pure-core (no vscode runtime import). The shellIntegration
 * adapter passes vscode's value in via duck typing.
 */
export function normalizeCommandLine(commandLine: { value: string; confidence: number }): string;

/**
 * Match against built-ins + custom patterns. Returns first agent whose any pattern
 * matches the (already normalized) command line, or null.
 *
 * Iteration order MUST be deterministic — built-ins first in BUILT_IN_PATTERNS
 * declaration order, then customPatterns. Custom patterns associate with the
 * `customAgentName` key passed alongside in buildMatcher().
 */
export function matchAgentCommand(
  normalized: string,
  customMatchers?: Array<{ agent: AgentName; pattern: RegExp }>,
): AgentMatch | null;

/**
 * Compose built-ins with user-supplied detect.customPatterns (DET-10).
 * Each user pattern is auto-prefixed with `^` (no escaping — user-controlled).
 *
 * customPatterns shape: Record<agentName, string[]> — keys become the agent
 * label that flows to {agent} templating; values are raw regex source strings.
 *
 * Example input: { "claude-next": ["claude-next\\b"] } → matcher recognizes
 * `claude-next chat` and emits agent="claude-next".
 *
 * If a user provides an invalid regex, that single pattern is silently dropped
 * (D-18). The rest of the matcher continues working.
 */
export function buildMatcher(
  customPatterns?: Record<string, string[]>,
): (normalized: string) => AgentMatch | null;
```

**Strip pipeline (locked from CONTEXT.md DET-09 ordering + RESEARCH.md Code Examples):**

```typescript
export const ANSI_CSI_RE = /\u001B\[[0-?]*[ -/]*[@-~]/g;
const PROMPT_PREFIX_RE = /^(\[[^\]]*\]\s*)?[$%❯→▶]\s*|^PS\s+[A-Za-z]:\\[^>]*>\s*/;

export function stripAndNormalize(raw: string): string {
  return raw
    .replace(ANSI_CSI_RE, "")          // 1. Strip CSI sequences
    .replace(/\r?\n/g, " ")            // 2. Normalize line endings to space
    .trimStart()                       // 3. Trim leading whitespace
    .replace(PROMPT_PREFIX_RE, "")     // 4. Strip prompt prefix
    .replace(/\s{2,}/g, " ")           // 5. Collapse multiple spaces
    .trim();                           // 6. Final trim
}
```

Note the PROMPT_PREFIX_RE alternation: it covers `[user@host path] $`, `%`, `❯`, `→`, `$`, `▶` AND PowerShell-style `PS C:\path>`.

<pitfalls>
- **Pure-core boundary:** Do NOT `import * as vscode from "vscode"` or `import { TerminalShellExecutionCommandLineConfidence } from "vscode"`. Use bare `number` for the confidence param. Tests pass numeric literals (0/1/2) directly. RESEARCH §Pattern 3 confirms enum values.
- **First-word anchoring on user patterns:** auto-prepend `^` even if the user already wrote it (`^^foo` is still a valid anchored regex; double-anchor is a no-op).
- **Invalid regex from user → silent drop, not throw:** wrap `new RegExp(...)` in try/catch (D-18).
- **Env-prefix vs argument-position confusion:** `AGENT_CLIENT_ID=x claude` matches (env-prefix is "still claude command"). `git commit -m "fix claude"` does NOT match (claude is a string argument). The prePeel helper handles this — only KEY=value\s+ patterns strip; everything else doesn't.
- **Pitfall 6 from RESEARCH (commandLine.value updates after start):** Not relevant here — we accept the Low-confidence value at start time. The strip pipeline absorbs noise.
- **Confidence enum numeric values are stable** (per @types/vscode 1.115.0) — using bare 0/1/2 is safer than re-importing the enum just to keep the module pure-core.
</pitfalls>
