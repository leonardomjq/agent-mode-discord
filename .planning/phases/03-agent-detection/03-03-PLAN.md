```typescript
// src/detectors/polling.ts
import * as vscode from "vscode";
import type { Event } from "../state/types";

export interface PollingDetectorOptions {
  /**
   * detect.polling.terminalNamePatterns — array of regex SOURCE STRINGS supplied by user.
   * Empty array (default) → polling disabled entirely (DET-06 zero-false-positive).
   * Each pattern is auto-anchored with ^ on the matcher; user writes `Claude Code` to match terminal named "Claude Code".
   */
  patterns?: string[];

  /**
   * Defaults to "claude" — the agent label dispatched when a terminal name matches.
   * Phase 4 may extend with per-pattern agent labels; v0.1 fixed.
   */
  defaultAgent?: string;

  /** Override for tests. Defaults to 5000 ms. */
  intervalMs?: number;

  /** Override for tests — clock injection. Defaults to setInterval. */
  setInterval?: typeof globalThis.setInterval;
  clearInterval?: typeof globalThis.clearInterval;

  /** Override for tests — terminals list source. Defaults to () => vscode.window.terminals. */
  getTerminals?: () => readonly vscode.Terminal[];
}

export interface PollingDetector {
  readonly tier: 4;
  start(dispatch: (event: Event) => void): vscode.Disposable;
}

export function createPollingDetector(opts?: PollingDetectorOptions): PollingDetector;
```

<pitfalls>
- **Pitfall 5 (RESEARCH):** Empty patterns array MUST short-circuit before any setInterval construction. Failure mode if you skip: the polling tier registers an interval that runs every 5s for the extension's whole lifetime even though no patterns are configured — wasted CPU, especially compounded across hundreds of users with default config.
- **Aggregate dispatch:** Do NOT dispatch one agent-started per matching terminal. The orchestrator (Plan 03-04) handles per-terminal aggregation; polling contributes a single "any terminal name matches" signal. If you fire 3 agent-starts for 3 matching terminals, the orchestrator double-counts and may double-down-cycle on agent-ended.
- **Auto-anchoring:** Same pattern as regex.ts (DET-10) — auto-prepend `^`. User writes `Claude` to match terminal named "Claude Session"; doesn't need to write `^Claude`.
- **Invalid user regex → silent drop (D-18):** Wrap `new RegExp("^" + p)` in try/catch. Don't throw on bad input; users may have typos. Rest of patterns continue to work.
- **Closed terminal not in vscode.window.terminals:** When a terminal is disposed, it's removed from the array. Detect by iterating the active set and checking inclusion in current terminals; if missing, remove from active set + maybe dispatch agent-ended.
- **No coordination with other tiers HERE:** Plan 03-04 orchestrator handles "lower tier silenced when higher tier active". Polling.ts dispatches naively; the orchestrator dedupes.
</pitfalls>
