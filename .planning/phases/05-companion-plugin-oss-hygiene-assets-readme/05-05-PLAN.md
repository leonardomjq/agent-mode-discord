**Rewrite `.github/workflows/ci.yml`** (per D-08 — expand existing, don't create new):

```yaml
name: CI

on:
  pull_request:
  push:
    branches: [main]

jobs:
  build:
    runs-on: ${{ matrix.os }}
    strategy:
      matrix:
        os: [ubuntu-latest, macos-latest, windows-latest]
      fail-fast: false
    steps:
      - uses: actions/checkout@v4

      - name: Setup pnpm
        uses: pnpm/action-setup@v4
        with:
          version: 9

      - name: Setup Node
        uses: actions/setup-node@v4
        with:
          node-version: 20
          cache: pnpm

      - name: Install
        run: pnpm install --frozen-lockfile

      - name: Lint
        run: pnpm lint

      - name: Build
        run: pnpm build

      - name: Bundle size check
        run: pnpm check:bundle-size

      - name: API surface check
        run: pnpm check:api-surface

      - name: Check ≤20 configuration keys (CONF-01)
        run: pnpm check:config-keys

      - name: Check zero outbound HTTP in built bundle (PRIV-07)
        run: pnpm check:no-network

      - name: Test
        run: pnpm test
```

Key changes from current:
1. Added `strategy.matrix.os` with 3 OS values (DIST-06)
2. Changed `runs-on` from `ubuntu-latest` to `${{ matrix.os }}`
3. Added `fail-fast: false` (per T-05-02 — don't cancel all if Windows flakes)
4. Added `pnpm lint` step before build (per D-08, D-09)
5. Kept all existing check steps (bundle-size, api-surface, config-keys, no-network)
6. Step order: install → lint → build → checks → test (lint before build catches type errors early; checks after build because they need dist/extension.cjs)
