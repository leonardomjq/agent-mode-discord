```typescript
import { readFileSync, readdirSync } from "node:fs";
import { join } from "node:path";

describe("REQ-4 guardrail: literal 'Agent Mode' purged from payload paths", () => {
  it("no source file under src/presence/ contains the literal '\"Agent Mode\"'", () => {
    const dir = join(__dirname, "..", "src", "presence");
    const files = readdirSync(dir).filter((f) => f.endsWith(".ts"));
    const offenders: string[] = [];
    for (const f of files) {
      const content = readFileSync(join(dir, f), "utf8");
      // Match double-quoted "Agent Mode" — preserves the SPEC's literal-string ban.
      // Allows mentions in comments to remain (comments are exempt — REQ-4 targets
      // the runtime payload string only). Refine if a comment hit ever shows up.
      const lines = content.split("\n");
      lines.forEach((line, idx) => {
        // Strip line comments to keep the guardrail focused on runtime strings.
        const codeOnly = line.replace(/\/\/.*$/, "");
        if (/"Agent Mode"/.test(codeOnly)) {
          offenders.push(`${f}:${idx + 1} → ${line.trim()}`);
        }
      });
    }
    expect(offenders).toEqual([]);
  });
});
```

This guardrail is intentionally narrow:
- Restricted to `src/presence/` (the payload-emitting directory).
- Strips line comments (`// ...`) before testing — REQ-4 targets the runtime card surface, not historical comments. (Block comments `/* ... */` are not stripped here; if a future block comment contains the literal it must be edited or the regex narrowed further.)
- Allows REQ-5's `src/outputChannel.ts` to retain `"Agent Mode (Discord)"` as the internal log channel name — that file lives in `src/`, not `src/presence/`, and is excluded by the directory scoping.
