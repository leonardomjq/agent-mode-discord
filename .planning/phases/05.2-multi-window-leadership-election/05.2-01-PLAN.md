```typescript
export interface LeadershipFsSurface {
  open: (path: string, flags: string) => Promise<{ close: () => Promise<void> }>;
  utimes: (path: string, atime: number | Date, mtime: number | Date) => Promise<void>;
  unlink: (path: string) => Promise<void>;
  stat: (path: string) => Promise<{ mtimeMs: number }>;
}

export interface LeadershipOptions {
  lockfilePath?: string;        // default: path.join(os.homedir(), ".claude", "agent-mode-discord.leader.lock")  (D-01)
  heartbeatMs?: number;         // default: 30_000  (D-03)
  stalenessMs?: number;         // default: 90_000  (D-03, 3× heartbeat)
  fs?: LeadershipFsSurface;     // default: node:fs/promises wrappers
  now?: () => number;           // default: Date.now
  setInterval?: (fn: () => void, ms: number) => NodeJS.Timeout;
  clearInterval?: (handle: NodeJS.Timeout) => void;
  log?: (msg: string) => void;  // optional verbose log sink (D-13)
}

export interface Leadership {
  acquire(): Promise<boolean>;    // true = became leader; false = already held by fresh leader
  release(): Promise<void>;       // unlinks lockfile if isLeader; silent on failure (D-07)
  isLeader(): boolean;
  onTakeover(cb: () => void): void; // fires when a follower successfully takes over a stale leader (D-04)
}

export function createLeadership(opts?: LeadershipOptions): Leadership;
```
