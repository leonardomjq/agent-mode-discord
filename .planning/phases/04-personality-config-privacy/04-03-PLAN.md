    Create `src/presence/templater.ts` as pure-core (no vscode import). Implementation:

    ```ts
    export interface TemplateTokens {
      workspace?: string;
      filename?: string;
      language?: string;
      branch?: string;
      agent?: string;
      elapsed?: string;
    }

    const TOKEN_RE = /\{(\w+)\}/g;

    export function renderTemplate(message: string, tokens: TemplateTokens): string {
      return message.replace(TOKEN_RE, (_match, key: string) => {
        const val = (tokens as Record<string, string | undefined>)[key];
        return typeof val === "string" ? val : "";
      });
    }

    export function isBlank(s: string): boolean {
      return s.trim().length === 0;
    }
    ```

    **Note on blank-skip semantics (from 04-RESEARCH.md Pitfall 2):** `isBlank` returns true ONLY for fully-whitespace output. Messages like `"{workspace} · {branch}"` with both tokens empty render as `" · "` which is NOT blank — that's intentional; a bullet separator is still visible copy. The animator's skip-blank logic should only skip strictly-empty results, with a 10-attempt cap before hard-fallback to `"building, afk"` (plan 04-02 owns that logic).
