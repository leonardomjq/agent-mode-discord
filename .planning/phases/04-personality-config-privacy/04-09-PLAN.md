Document the runtime intercept as a future stretch in the plan summary.

    Create `scripts/check-no-network.mjs`:

    ```js
    #!/usr/bin/env node
    import { readFileSync, existsSync } from "node:fs";

    const target = process.argv[2] ?? "dist/extension.cjs";

    if (!existsSync(target)) {
      console.error(`FAIL: ${target} not found — run \`pnpm build\` first.`);
      process.exit(1);
    }

    const src = readFileSync(target, "utf8");

    // FORBIDDEN: exact list of outbound-HTTP surface references (reviewer R2).
    // Any match is a FAIL — no WARN path. Rationale for each entry:
    // - http.request(/https.request( : direct Node core HTTP calls
    // - require/import of http/https/node:http/node:https : loading the HTTP stdlib
    // - require/import of undici/got/axios/node-fetch : known HTTP libs
    // - new XMLHttpRequest / globalThis.fetch / fetch( : web-style HTTP
    const FORBIDDEN = [
      { re: /\bhttp\.request\s*\(/g,                         reason: "http.request() call" },
      { re: /\bhttps\.request\s*\(/g,                        reason: "https.request() call" },
      { re: /\brequire\(["'](node:)?http["']\)/g,             reason: "require('http') / require('node:http')" },
      { re: /\brequire\(["'](node:)?https["']\)/g,            reason: "require('https') / require('node:https')" },
      { re: /\bfrom\s+["'](node:)?https?["']/g,                reason: "import from 'http' / 'https' / 'node:http' / 'node:https'" },
      { re: /\brequire\(["']undici["']\)/g,                   reason: "require('undici')" },
      { re: /\bfrom\s+["']undici["']/g,                        reason: "import from 'undici'" },
      { re: /\brequire\(["']got["']\)/g,                      reason: "require('got')" },
      { re: /\bfrom\s+["']got["']/g,                           reason: "import from 'got'" },
      { re: /\brequire\(["']axios["']\)/g,                    reason: "require('axios')" },
      { re: /\bfrom\s+["']axios["']/g,                         reason: "import from 'axios'" },
      { re: /\brequire\(["']node-fetch["']\)/g,               reason: "require('node-fetch')" },
      { re: /\bfrom\s+["']node-fetch["']/g,                    reason: "import from 'node-fetch'" },
      { re: /\bnew\s+XMLHttpRequest\b/g,                      reason: "new XMLHttpRequest" },
      { re: /\bglobalThis\.fetch\b/g,                         reason: "globalThis.fetch access" },
      { re: /\bfetch\s*\(/g,                                  reason: "fetch( call — reviewer R2 promoted WARN→FAIL" },
    ];

    // IPC exemption documentation (reviewer R2):
    // @xhayper/discord-rpc uses net.createConnection({ path: "/tmp/discord-ipc-N" })
    // on POSIX and \\?\\pipe\\discord-ipc-N on win32. That call is NOT in FORBIDDEN
    // because it does not open a TCP/HTTP socket. We sweep for the TCP variant
    // (net.createConnection / net.connect called with a `host:` literal nearby)
    // and treat that as WARN — requires manual review. The FAIL gate remains
    // strictly the FORBIDDEN list above.
    const TCP_NET_HEURISTIC = /\bnet\.(createConnection|connect)\s*\([^)]{0,120}\bhost\s*:/g;

    let failed = false;
    for (const { re, reason } of FORBIDDEN) {
      const matches = src.match(re);
      if (matches && matches.length > 0) {
        // Surrounding-context sample for CI logs: first 80 chars around the first match.
        const idx = src.search(re);
        const sample = src.slice(Math.max(0, idx - 20), Math.min(src.length, idx + 60)).replace(/\s+/g, " ");
        console.error(`FAIL: ${matches.length}× ${reason} — ${re} in ${target}. Sample: ...${sample}...`);
        failed = true;
      }
    }

    const tcpMatches = src.match(TCP_NET_HEURISTIC);
    if (tcpMatches && tcpMatches.length > 0) {
      console.warn(`WARN: ${tcpMatches.length}× net.createConnection/connect() co-located with host: literal — verify this is NOT a TCP HTTP call. IPC (path:) is fine.`);
    }

    if (failed) {
      console.error(`\nPRIV-07 violation: bundle contains forbidden HTTP surface. See matches above.`);
      process.exit(1);
    }
    console.log(`PASS: zero outbound HTTP patterns in ${target} (${src.length} bytes scanned)`);
    ```

If a forbidden pattern appears, SHARPEN the regex (already done — every entry requires `\s*\(` where applicable) or remove the offending import from source. Never loosen the guard.
