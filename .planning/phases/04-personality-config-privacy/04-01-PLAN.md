export type Message = string | string[];

export interface Pack {
  id: string;
  version: 1;
  pools: {
    AGENT_ACTIVE: { _primary: Message[]; [agent: string]: Message[] };
    CODING: Message[];
    IDLE: Message[];
  };
  timeOfDay?: {
    lateNight?: Message[];
    morning?: Message[];
    afternoon?: Message[];
    evening?: Message[];
  };
}

export type ValidateResult =
  | { ok: true; pack: Pack }
  | { ok: false; error: string };

// src/presence/packLoader.ts
export interface PackLoaderDeps {
  readFile: (p: string) => string;          // throws on ENOENT
  stat: (p: string) => { size: number };    // throws on ENOENT
  now: () => Date;
  log: (line: string) => void;              // debug channel
}
export const realPackLoaderDeps: PackLoaderDeps;
export function validatePack(raw: unknown): ValidateResult;
export function loadPack(
  opts: { customPackPath: string; builtin: Pack },
  deps?: Partial<PackLoaderDeps>,
): Pack;

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| user config → fs.readFileSync | customPackPath is user-supplied; untrusted file content |
| JSON.parse → validator | untrusted parsed object; may carry `__proto__` / weird shapes |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-01 | Denial of Service | loadPack fs read | mitigate | `stat(path).size > 100_000` → reject + debug-log; never read > 100 KB file |
| T-04-02 | Tampering (proto pollution) | validatePack | mitigate | Validator enumerates known keys only; never spreads raw input; returns a fresh narrowed object; JSON.parse() inherently does not re-apply inherited props |
| T-04-03 | Tampering (ReDoS) | validatePack | accept | Validator contains no regex against user input; only shape checks. No ReDoS surface in this plan. |
</threat_model>

    Implement `loadPack(opts, depsOverride = {})`:
    ```ts
    export function loadPack(
      opts: { customPackPath: string; builtin: Pack },
      depsOverride: Partial<PackLoaderDeps> = {},
    ): Pack {
      const deps = { ...realPackLoaderDeps, ...depsOverride };
      const { customPackPath, builtin } = opts;
      if (!customPackPath) return builtin;
      try {
        const { size } = deps.stat(customPackPath);
        if (size > 100_000) {
          deps.log(`[packLoader] custom pack too large (${size} > 100000): ${customPackPath} — falling back to built-in`);
          return builtin;
        }
        const raw = deps.readFile(customPackPath);
        let parsed: unknown;
        try { parsed = JSON.parse(raw); }
        catch (err) {
          deps.log(`[packLoader] custom pack JSON parse error: ${String(err)} — falling back`);
          return builtin;
        }
        const result = validatePack(parsed);
        if (!result.ok) {
          deps.log(`[packLoader] custom pack schema invalid: ${result.error} — falling back`);
          return builtin;
        }
        return result.pack;
      } catch (err) {
        deps.log(`[packLoader] custom pack read failed: ${String(err)} — falling back`);
        return builtin;
      }
    }
    ```
